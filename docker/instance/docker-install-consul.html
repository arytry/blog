# dockerе®үиЈ…consul

жҸҗзӨә

иҝӣе…Ҙconsulе®№еҷЁзҡ„е‘Ҫд»ӨжҳҜ/bin/sh

docker exec -it consul /bin/sh

з«ҜеҸЈиҜҙжҳҺ

  • 8500 http з«ҜеҸЈпјҢз”ЁдәҺ http жҺҘеҸЈе’Ң web ui
  • 8300 server rpc з«ҜеҸЈпјҢеҗҢдёҖж•°жҚ®дёӯеҝғ consul server д№Ӣй—ҙйҖҡиҝҮиҜҘз«ҜеҸЈйҖҡдҝЎ
  • 8301 serf lan з«ҜеҸЈпјҢеҗҢдёҖж•°жҚ®дёӯеҝғ consul client йҖҡиҝҮиҜҘз«ҜеҸЈйҖҡдҝЎ
  • 8302 serf wan з«ҜеҸЈпјҢдёҚеҗҢж•°жҚ®дёӯеҝғ consul server йҖҡиҝҮиҜҘз«ҜеҸЈйҖҡдҝЎ
  • 8600 dns з«ҜеҸЈпјҢз”ЁдәҺжңҚеҠЎеҸ‘зҺ°

# еҚ•жңәеҚ•иҠӮзӮ№

docker run -d --restart=always -p 8500:8500 --name consul consul agent -server -bootstrap -ui -client='0.0.0.0'

  • -serverпјҡи®ҫзҪ®дёәServerиҠӮзӮ№жЁЎејҸпјҢй»ҳи®ӨдёәClientиҠӮзӮ№жЁЎејҸпјҢиҝҷдёӘе’Ң-clientеҸӮж•°жҳҜдёҚеҗҢзҡ„жҰӮеҝө

    consulжңүClientе’ҢServerдёӨз§ҚиҠӮзӮ№жЁЎејҸпјҢе®ғд»¬зҡ„еҢәеҲ«еңЁconsulе®ҳзҪ‘open in new windowдёӯжңүиҝҷд№ҲдёҖж®өиҜқпјҡ

    The only difference between servers and clients are that servers are the only components that store and replicate data.

    е°ұжҳҜиҜҙServerиҠӮзӮ№жҜ”ClientиҠӮзӮ№еӨҡдәҶеӯҳеӮЁе’ҢеӨҚеҲ¶ж•°жҚ®зҡ„иғҪеҠӣ

    дёҚз®ЎжҳҜServerиҝҳжҳҜClientпјҢз»ҹз§°дёәAgent

    ClientжҳҜзӣёеҜ№ж— зҠ¶жҖҒзҡ„пјҢеҸӘиҙҹиҙЈиҪ¬еҸ‘RPCеҲ°ServerпјҢиө„жәҗејҖй”ҖеҫҲе°‘

    ServerжҳҜдёҖдёӘжңүдёҖз»„жү©еұ•еҠҹиғҪзҡ„д»ЈзҗҶпјҢиҝҷдәӣеҠҹиғҪеҢ…жӢ¬еҸӮдёҺRaftйҖүдёҫпјҢз»ҙжҠӨйӣҶзҫӨзҠ¶жҖҒпјҢе“Қеә”RPCжҹҘиҜўпјҢдёҺе…¶д»–ж•°жҚ®дёӯеҝғдәӨдә’WAN Gossipе’ҢиҪ¬еҸ‘жҹҘиҜўз»ҷleaderжҲ–иҖ…иҝңзЁӢж•°жҚ®дёӯеҝғ

    еҸӘжңүServerиҠӮзӮ№жүҚиғҪиў«йҖүдёҫдёәleader

    д»ҘдёӢеҸӮж•°йңҖиҰҒе’Ң-serverй…ҚеҗҲдҪҝз”ЁпјҢеҸҜд»ҘдёҚи®ҫзҪ®

    • -bootstrapпјҡеҗҜеҠЁжЁЎејҸпјҢжӯӨжЁЎејҸдёӢпјҢиҠӮзӮ№еҸҜд»ҘйҖүдёҫиҮӘе·ұдёәleaderпјҢдёҖдёӘdatacenterеҸӘиғҪжңүдёҖдёӘжӯӨжЁЎејҸеҗҜеҠЁзҡ„иҠӮзӮ№

      еҚ•иҠӮзӮ№йңҖиҰҒеҗҜз”ЁпјҢйӣҶзҫӨеҗҜеҠЁеҗҺпјҢж–°еҗҜеҠЁзҡ„иҠӮзӮ№дёҚе»әи®®дҪҝз”Ёиҝҷз§ҚжЁЎејҸ

    • -bootstrap-expectпјҡи®ҫе®ҡдёҖдёӘж•°жҚ®дёӯеҝғйңҖиҰҒзҡ„serverиҠӮзӮ№ж•°пјҢи®ҫзҪ®зҡ„ж•°еӯ—еҝ…йЎ»е’Ңе®һйҷ…зҡ„serverиҠӮзӮ№ж•°еҢ№й…Қ

      consulдјҡзӯүеҫ…зӣҙеҲ°ж•°жҚ®дёӯеҝғдёӢзҡ„serverиҠӮзӮ№ж»Ўи¶іи®ҫе®ҡжүҚдјҡеҗҜеҠЁйӣҶзҫӨжңҚеҠЎе№¶йҖүдёҫleader

      дёҚиғҪе’Ң-bootstrapж··з”Ё

  • -ui еҗҜз”Ё UI з•Ңйқў

  • -client жҢҮе®ҡclientиҠӮзӮ№и®ҝй—®зҡ„IPпјҢй»ҳи®ӨжҳҜ127.0.0.1пјҢ0.0.0.0иЎЁзӨәдёҚйҷҗе®ўжҲ·з«ҜIP

  • -retry-join жҢҮе®ҡиҰҒеҠ е…Ҙзҡ„иҠӮзӮ№ең°еқҖпјҲз»„е»әйӣҶзҫӨпјүпјҢеҗҺйқўеҸҜд»Ҙи·ҹе…¶е®ғиҠӮзӮ№зҡ„ipпјҢпјҲеҚ•жңәпјүд№ҹеҸҜд»Ҙи·ҹиҠӮзӮ№еҗҚз§°

    жіЁж„Ҹ

    -join е·Із»Ҹиў«ејғз”ЁпјҢдҪҝз”Ё-retry-joinжӣҝд»Ј

  • -node жҢҮе®ҡиҠӮзӮ№еҗҚз§°

  • -data-dirпјҡжҢҮе®ҡж•°жҚ®еӯҳж”ҫзӣ®еҪ•пјҲиҜҘзӣ®еҪ•еҝ…йңҖеӯҳеңЁпјҢйңҖжҸҗеүҚеҲӣе»әеҘҪпјү

  • -config-dirпјҡжҢҮе®ҡй…ҚзҪ®ж–Үд»¶зӣ®еҪ•

  • -bindпјҡз»‘е®ҡзҡ„еҶ…йғЁйҖҡи®Ҝең°еқҖпјҢй»ҳи®Ө0.0.0.0пјҢеҚіпјҢжүҖжңүзҡ„жң¬ең°ең°еқҖпјҢдјҡе°Ҷз¬¬дёҖдёӘеҸҜз”Ёзҡ„ipең°еқҖж•Јж’ӯеҲ°йӣҶзҫӨдёӯпјҢеҰӮжһңжңүеӨҡдёӘеҸҜз”Ёзҡ„ipv4пјҢеҲҷconsulеҗҜеҠЁжҠҘй”ҷгҖӮ[::]ipv6пјҢTCP UDPеҚҸи®®пјҢзӣёеҗҢзҡ„з«ҜеҸЈгҖӮйҳІзҒ«еўҷй…ҚзҪ®гҖӮ

# еҚ•жңәеӨҡиҠӮзӮ№

version: '3'

services:
  cs1: 
    image: consul
    command: agent -server -client=0.0.0.0 -bootstrap-expect=3 -node=cs1 -data-dir=/var/opt/consul
    volumes:
      - /srv/consul/cs1/etc:/etc/consul
      - /srv/consul/cs1/opt:/var/opt/consul
  cs2:
    image: consul
    command: agent -server -client=0.0.0.0 -retry-join=cs1 -node=cs2 -data-dir=/var/opt/consul
    volumes:
      - /srv/consul/cs2/etc:/etc/consul
      - /srv/consul/cs2/opt:/var/opt/consul
    depends_on:
      - cs1
  cs3:
    image: consul
    command: agent -server -client=0.0.0.0 -retry-join=cs1 -node=cs3 -data-dir=/var/opt/consul
    volumes:
      - /srv/consul/cs3/etc:/etc/consul
      - /srv/consul/cs3/opt:/var/opt/consul
    depends_on:
      - cs1
  cc1:
    image: consul
    command: agent -client=0.0.0.0 -retry-join=cs1 -ui -node=cc1 -data-dir=/var/opt/consul  
    ports:
      - 8500:8500
    volumes:
      - /srv/consul/cc1/etc:/etc/consul
      - /srv/consul/cc1/opt:/var/opt/consul
    depends_on:
      - cs2
      - cs3
