# dockerе®үиЈ…rabbitmq

жҸҗзӨә

д»ҘдёӢеҶ…е®№йңҖиҰҒжңүдёҖе®ҡзҡ„dockerеҹәзЎҖпјҢе№¶дё”йңҖиҰҒеңЁе·Із»Ҹе®үиЈ…дәҶdockerзҺҜеўғдёӯиҝҗиЎҢ

# е®үиЈ…

docker pull rabbitmq

иҝҷж ·й»ҳи®ӨжӢүеҸ–зҡ„жҳҜlatestзүҲжң¬пјҢиҜҘзүҲжң¬й»ҳи®ӨжІЎжңүз®ЎзҗҶз•ҢйқўпјҢеҰӮжһңйңҖиҰҒз®ЎзҗҶз•ҢйқўжҲ‘д»¬еҸҜд»Ҙжһ„е»әе®№еҷЁеҗҺеңЁе®№еҷЁдёӯеҗҜз”ЁжҺ§еҲ¶еҸ°жҸ’д»¶

жҲ–иҖ…зӣҙжҺҘжӢүеҸ–managementзүҲжң¬

docker pull rabbitmq:management

# иҝҗиЎҢ

# жңҖе°ҸеҢ–жһ„е»ә

й»ҳи®ӨlatestзүҲжң¬

docker run --name=rabbitmq -p 15672:15672 -p 5672:5672 -d rabbitmq
  • --name=rabbitmq е®№еҷЁе‘ҪеҗҚдёәrabbitmq
  • -p 15672:15672 е°Ҷе®№еҷЁзҡ„15672з«ҜеҸЈжҳ е°„еҲ°е®ҝдё»жңәзҡ„15672з«ҜеҸЈпјҢ15672жҳҜз®ЎзҗҶз•Ңйқўз«ҜеҸЈ
  • -p 5672:5672 е°Ҷе®№еҷЁзҡ„5672з«ҜеҸЈжҳ е°„еҲ°е®ҝдё»жңәзҡ„5672з«ҜеҸЈпјҢ5672жҳҜжңҚеҠЎз«ҜеҸЈ

еҰӮжһңжҲ‘д»¬жӢүеҸ–зҡ„жҳҜmanagementзүҲжң¬пјҢи®°еҫ—еңЁе®№еҷЁеҗҺйқўеҠ дёҠmanagementж Үзӯҫ

docker run --name=rabbitmq -p 15672:15672 -p 5672:5672 -d rabbitmq:management

жіЁж„Ҹ

еҰӮжһңжһ„е»әе®№еҷЁж—¶жІЎжңүеҠ managementж ҮзӯҫпјҢзі»з»ҹдјҡй»ҳи®ӨжӢүеҸ–latestзүҲжң¬еҗҺеҶҚиҝҗиЎҢ

еҰӮжһңжҳҜйҖҡиҝҮmanagementж Үзӯҫжһ„е»әзҡ„е®№еҷЁпјҢзҺ°еңЁжҲ‘д»¬е°ұеҸҜд»Ҙи®ҝй—®http://е®ҝдё»жңәip:15672жү“ејҖз®ЎзҗҶжҺ§еҲ¶еҸ°з•ҢйқўдәҶпјҢеҪ“з„¶еҰӮжһңзҺҜеўғпјҲйҳҝйҮҢдә‘пјүжңүз«ҜеҸЈйҷҗеҲ¶иҝҳйңҖиҰҒе…ҲеҗҜз”Ё15672з«ҜеҸЈжүҚиЎҢ

# жҺ§еҲ¶еҸ°жҸ’д»¶

еҰӮжһңжҳҜз”Ёй»ҳи®ӨlatestзүҲжң¬жһ„е»әе®№еҷЁпјҢд№ҹжҳҜеҸҜд»Ҙжү“ејҖжҺ§еҲ¶еҸ°з•Ңйқўзҡ„пјҢиҝӣе…Ҙе®№еҷЁйҖҡиҝҮе‘Ҫд»ӨеҗҜз”ЁжҺ§еҲ¶еҸ°жҸ’д»¶пјҢз„¶еҗҺйҮҚеҗҜе®№еҷЁеҚіеҸҜ

# иҝӣе…Ҙе®№еҷЁ
docker exec -it rabbitmq /bin/bash

# еҗҜз”ЁжҺ§еҲ¶еҸ°жҸ’д»¶
rabbitmq-plugins enable rabbitmq_management

# йҖҖеҮәе®№еҷЁ
exit

# йҮҚеҗҜе®№еҷЁ
docker restart rabbitmq

# еҲқе§ӢиҙҰеҸ·

еҰӮжһңиҝҗиЎҢе®№еҷЁж—¶жІЎжңүи®ҫзҪ®иҙҰеҸ·пјҢrabbitmqдјҡз»ҷжҲ‘д»¬еҲӣе»әдёҖдёӘй»ҳи®ӨиҙҰеҸ·пјҡguest/guest

еҰӮжһңдёҚжғіз”Ёй»ҳи®Өзҡ„guestпјҢеҸҜд»ҘеңЁжһ„е»әе®№еҷЁж—¶жҢҮе®ҡе…¶е®ғиҙҰеҸ·

docker run --name=rabbitmq \
    -p 15672:15672 -p 5672:5672 \
    --restart=always \
    -e RABBITMQ_DEFAULT_USER=admin \
    -e RABBITMQ_DEFAULT_PASS=admin \
    -d rabbitmq
  • -e RABBITMQ_DEFAULT_USER=admin й»ҳи®ӨиҙҰеҸ·дёәadmin
  • -e RABBITMQ_DEFAULT_PASS=admin й»ҳи®ӨеҜҶз Ғдёәadmin

жҸҗзӨә

зҷ»еҪ•иҝӣе…ҘжҺ§еҲ¶еҸ°еҗҺпјҢд№ҹеҸҜд»ҘйҖҡиҝҮAdminиҸңеҚ•ж·»еҠ гҖҒжӣҙж”№иҙҰжҲ·дҝЎжҒҜ

# е»¶иҝҹйҳҹеҲ—

# дёӢиҪҪжҸ’д»¶

# еёёи§Ғй—®йўҳ

зӮ№еҮ»ChannelsиҸңеҚ•пјҢжҸҗзӨәStats in management UI are disabled on this node

жҲ–иҖ…

зӮ№еҮ»ExchangesиҸңеҚ•пјҢйҡҸж„ҸзӮ№еҮ»дёҖдёӘexchange nameжҸҗзӨәManagement API returned status code 500 -

и§ЈеҶіеҠһжі•пјҡ иҝӣе…Ҙе®№еҷЁпјҢиҝӣе…Ҙ/etc/rabbitmq/conf.dзӣ®еҪ•пјҢдҝ®ж”№management_agent.disable_metrics_collector.confж–Үд»¶дёӯзҡ„management_agent.disable_metrics_collector = falseпјҢз„¶еҗҺйҖҖеҮәйҮҚеҗҜе®№еҷЁеҚіеҸҜ
